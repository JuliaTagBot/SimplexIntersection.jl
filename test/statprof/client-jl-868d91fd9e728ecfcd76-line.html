232 function incomplete_tag(ex::Expr)
233 Meta.isexpr(ex, :incomplete) || return :none
234 msg = ex.args[1]
235 contains(msg, "string") && return :string
236 contains(msg, "comment") && return :comment
237 contains(msg, "requires end") && return :block
